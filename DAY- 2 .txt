          Scanner sc = new Scanner(System.in);
        int a = sc.nextInt();
        int b=sc.nextInt();
        char c= sc.next().charAt(0);
        switch(c)
        {
                
            case '+':
                System.out.println(a+b);
                break;
                case '-':
                System.out.println(a-b);
                break;
                case '*':
                System.out.println(a*b);
                break;
                case '/':
                {
                    if(b!=0)
                        System.out.println(a/b);
                    else
                        System.out.println("Error");
                }
                break;
            default:
                System.out.println("operator out of bound");
                
        }
        
        
        
        
    

    }
    
}
